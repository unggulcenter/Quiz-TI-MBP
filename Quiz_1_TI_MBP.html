    function startQuiz() {

        const name = document.getElementById('userName').value;

        const npm = document.getElementById('userNPM').value;


        if (!name || !npm) {

        }


        // Simpan info user

        document.getElementById('header-info').innerText = `Peserta: ${name} | NPM: ${npm}`;


        // Acak dan Ambil 20 Soal

        selectedQuestions = bankSoal.sort(() => 0.5 - Math.random()).slice(0, 20);

        

        displayQuestions();


        document.getElementById('setup').classList.add('hidden');

        document.getElementById('quiz-area').classList.remove('hidden');

        window.scrollTo(0,0);

    }


    function displayQuestions() {

        const container = document.getElementById('questions-container');

        container.innerHTML = "";

        selectedQuestions.forEach((item, index) => {

            const qDiv = document.createElement('div');

            qDiv.className = 'question';

            qDiv.innerHTML = `

                <p><strong>Soal ${index + 1}:</strong> ${item.q}</p>

                <div class="options">

                    <label><input type="radio" name="q${index}" value="a"> a. ${item.a}</label>

                    <label><input type="radio" name="q${index}" value="b"> b. ${item.b}</label>

                    <label><input type="radio" name="q${index}" value="c"> c. ${item.c}</label>

                    <label><input type="radio" name="q${index}" value="d"> d. ${item.d}</label>

                </div>

            `;

            container.appendChild(qDiv);

        });

    }


    function calculateScore() {

        let correctCount = 0;

        selectedQuestions.forEach((item, index) => {

            const answer = document.querySelector(`input[name="q${index}"]:checked`);

            if (answer && answer.value === item.correct) {

                correctCount++;

            }

        });


        const score = (correctCount / 20) * 100;

        const name = document.getElementById('userName').value;


        document.getElementById('quiz-area').classList.add('hidden');

        document.getElementById('result-area').classList.remove('hidden');

        

        document.getElementById('summary-text').innerText = `Selamat ${name}, kuis telah selesai!`;

        document.getElementById('final-score').innerText = score;

        document.getElementById('correct-count').innerText = `Jawaban Benar: ${correctCount} dari 20 soal.`;

        window.scrollTo(0,0);

    }
